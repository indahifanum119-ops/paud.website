  <script>

    // Mobile menu (simple)

    const mobileBtn = document.getElementById('mobileBtn');

    mobileBtn.addEventListener('click', ()=> alert('Untuk versi mobile, gunakan menu navigasi di update berikutnya.'))

 

    // Hero image click -> modal

    const heroImg = document.getElementById('heroImg');

    const modal = document.getElementById('modal');

    const modalImg = document.getElementById('modalImg');

    const closeModal = document.getElementById('closeModal');

    heroImg.addEventListener('click', ()=>{ modalImg.src = heroImg.src; modal.classList.remove('hidden'); modal.classList.add('flex'); });

    closeModal.addEventListener('click', ()=>{ modal.classList.add('hidden'); modal.classList.remove('flex'); });

    modal.addEventListener('click', (e)=>{ if(e.target === modal) { modal.classList.add('hidden'); modal.classList.remove('flex'); } });

 

    // Simple slider

    const slides = document.getElementById('slides');

    const prev = document.getElementById('prev');

    const next = document.getElementById('next');

    let current = 0;

    const total = slides.children.length;

    function updateSlider(){ slides.style.transform = `translateX(-${current * 100}%)`; }

    prev.addEventListener('click', ()=>{ current = (current -1 + total) % total; updateSlider(); });

    next.addEventListener('click', ()=>{ current = (current +1) % total; updateSlider(); });

    // click slide -> open modal

    Array.from(slides.children).forEach(img=> img.addEventListener('click', (e)=>{ modalImg.src = e.target.src; modal.classList.remove('hidden'); modal.classList.add('flex'); }));

 

    // Registration form -> save to localStorage + toast

    const regForm = document.getElementById('regForm');

    const toast = document.getElementById('toast');

    regForm.addEventListener('submit', (e)=>{

      e.preventDefault();

      const data = { orangtua: document.getElementById('orangtua').value, namaanak: document.getElementById('namaanak').value, usia: document.getElementById('usia').value, wa: document.getElementById('wa').value, time: new Date().toISOString() };

      const existing = JSON.parse(localStorage.getItem('paud_reg')) || [];

      existing.push(data);

      localStorage.setItem('paud_reg', JSON.stringify(existing));

      // show toast

      toast.classList.remove('hidden');

      setTimeout(()=> toast.classList.add('hidden'), 3500);

      regForm.reset();

    });

 

    // Auto slide every 5s

    setInterval(()=>{ current = (current +1) % total; updateSlider(); }, 5000);

 

    // Daftar button scroll to form and pulse

    document.getElementById('daftarBtn').addEventListener('click', ()=>{

      document.getElementById('contact').scrollIntoView({ behavior: 'smooth' });

      document.getElementById('daftarBtn').classList.add('animate-pulse');

      setTimeout(()=> document.getElementById('daftarBtn').classList.remove('animate-pulse'), 1200);

    });

 

  </script>
